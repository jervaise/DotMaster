# Commands to commit version 1.0.8 changes and create a tag

# Add all modified files
git add bootstrap.lua
git add CHANGELOG.md
git add README.md
git add DotMaster.toc

# Commit the changes
git commit -m "Bump version to 1.0.8"

# Create and push a tag for the new version
git tag -a v1.0.8 -m "Version 1.0.8"
git push origin v1.0.8

# Push the changes to the main branch
git push origin main

# Done! The local environment is now set to version 1.0.8
